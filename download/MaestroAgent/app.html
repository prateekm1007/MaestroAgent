cd maestro
./install.sh --with-ollama

# Or use Docker directly
docker run -d \
  -p 3000:3000 \
  -p 8000:8000 \
  -v maestro_data:/data \
  -e OPENAI_API_KEY="sk-…" \
  -e ANTHROPIC_API_KEY="sk-ant-…" \
  maestro/maestro:latest

# → Maestro running on http://localhost:3000
# → Sandbox on http://localhost:8000
# → Ollama auto-detected on port 11434
